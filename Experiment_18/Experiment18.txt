Experiment 18: 
Write a shell script using a while loop to read a number from the user and print numbers from 1 up to that number. #whileNnos.sh


whilenos.sh
#!/bin/bash
read -p "Enter a number: " n
i=1
while [ $i -le $n ]
do
    echo -n "$i "
    i=$((i+1))
done
